const { assert } = require('chai');
const _ = require('lodash');

describe("Submission tests", () => {
  
  function doTest(year, expected) {
    const actual = isLeapYear(year);
    assert.strictEqual(actual, expected, `Incorrect answer for year ${year}`);
  }
  
  describe("Fixed tests", function() {
    it("Leap years divisible by 4", () => {
      doTest(2020, true);
      doTest(1824, true);
      doTest(2152, true);
    });

    it("Leap years divisible by 400", () => {
      doTest(1600, true);
      doTest(2000, true);
      doTest(4000, true);
    });

    it("Non-leap years divisible by 100", () => {
      doTest(1800, false);
      doTest(1900, false);
      doTest(2100, false);
      doTest(2200, false);
    });

    it("Regular non-leap years", () => {
      doTest(1821, false);
      doTest(1942, false);
      doTest(2113, false);
      doTest(2254, false);
    });
  });  
  
  describe("Random tests", function() {
    it("Random tests", function() {
      let yearMul400  = [ 1600, 2000, 2400, 2800, 3200, 3600 ];
      let yearMul100  = yearMul400.flatMap(y => [ y+100, y+200, y+300 ]);
      let yearMul4    = yearMul100.map(y => y + _.random(1, 24) * 4);
      let yearNonMul4 = yearMul4.map(y => y + _.random(1, 3));
      
      let testCases =
          [...yearMul400, ...yearMul4].map(y => [y, true])
          .concat(
            [...yearMul100, ...yearNonMul4].map(y => [y, false]));

      for(let [year, expected] of _.shuffle(testCases)) {
        doTest(year, expected);
      }
    });
  });    
});
